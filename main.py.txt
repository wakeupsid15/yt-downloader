from flask import Flask, request, jsonify
from flask_cors import CORS
import yt_dlp

app = Flask(__name__)
CORS(app) # This allows your GitHub site to talk to this server

@app.route('/get_info')
def get_info():
    url = request.args.get('url')
    ydl_opts = {'quiet': True, 'noplaylist': True}
    
    try:
        with yt_dlp.YoutubeDL(ydl_opts) as ydl:
            info = ydl.extract_info(url, download=False)
            formats = []
            for f in info.get('formats', []):
                # Only grab formats that have both video and audio for easy downloading
                if f.get('vcodec') != 'none' and f.get('acodec') != 'none':
                    formats.append({
                        'res': f.get('format_note', f'{f.get("height")}p'),
                        'ext': f['ext'],
                        'url': f['url']
                    })
            return jsonify({'formats': formats[::-1]}) # Newest/Highest quality first
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == "__main__":
    app.run()
